\section{Cinematica}

\subsection{Leggi orarie}
\frmheader{Descrizione}{Formula}
\frm{Spostamento}{x(t) = x_0 + v_0 t + \frac{1}{2} a t^2}
\frm{Velocità}{v(t) = v_0 + a(t - t_0)}
\frm{Velocità in Posizione}{v^2(x) = v_0^2 + 2a(x - x_1)}
\frm{Media}{v = \frac{v_1 - v_0}{v_1 - t_0} \ \text{(analogo per accelerazione)}}
\frm{Velocità con accelerazione}{v(t) = v_0 + \int_{t_0}^{t_1} a(t) \ dt}

\subsection{Moto del proiettile}
\frmheader{Nome Formula}{Formula}
\frm{Posizione x}{x(t) = x_0 + (v_0 \cos\phi_0)t}
\frm{Posizione y}{y(t) = y_0 + (v_0 \sin\phi_0)t - \frac{1}{2}gt^2}
\frm{Modulo velocità}{|\vec{v}| = \sqrt{v_x^2 + v_y^2}}
\frm{Angolo iniziale}{\tan\phi_0 = \frac{v_y}{v_x}}
\frm{Gittata}{R = \frac{v_0^2}{g} \sin 2\phi_0}

\subsection{Moto circolare}
\frmheader{Nome Formula}{Formula}
\frm{Posizione x}{x = R \cos \theta}
\frm{Posizione y}{y = R \sin \theta}
\frm{Vettore posizione}{\vec{r} = R \cos\theta \hat{i} + R \sin\theta \hat{j}}
\frm{Velocità angolare}{\omega = \frac{v_s}{R}}
\frm{Angolo orario}{\theta(t) = \theta_0 + \omega(t - t_0)}
\frm{Accelerazione centripeta}{a_c = \omega^2 R = \frac{v_s^2}{R} = v_s \omega}
\frm{Accelerazione angolare}{\alpha = \frac{d\omega}{dt}}
\frm{Accelerazione tangenziale}{a_t = \alpha R}
\frm{Accelerazione totale}{a = \sqrt{a_c^2 + a_t^2} = R\sqrt{\omega^4 + \alpha^2}}



\section{Forze}
\frmheader{Nome Formula}{Formula}
\frm{2° Legge di Newton}{\vec{F} = m\vec{a}}
\frm{Forza peso}{\vec{P} = m\vec{g}}
\frm{Forza vettoriale}{\vec{F} = F\cos\theta + F\sin\theta}
\frm{Forza di attrito statica}{F_s \le \mu_s N}
\frm{Forza di attrito dinamica}{F_k = \mu_k N}
\frm{Forza elastica}{F = -k (l - l_0) \ \text{con $k$ costante elastica}}
\frm{Molle in parallelo}{k_{eq} = k_1 + k_2}
\frm{Molle in serie}{\frac{1}{k_{eq}} = \frac{1}{k_1} + \frac{1}{k_2}}
\frm{Oscillatore armonico}{x(t) = A \cos(\omega t + \phi) \ \text{con $\omega = \sqrt{\frac{k}{m}}$, $A$ Ampiezza, $\omega$ Pulsazione, $\phi$ Fase}}
\frm{Periodo oscillatore armonico}{T = \frac{2\pi}{\omega} = 2\pi \sqrt{\frac{m}{k}}}
\frm{Frequenza oscillatore armonico}{\nu = \frac{1}{T} = \frac{\omega}{2\pi} = \frac{1}{2\pi} \sqrt{\frac{k}{m}}}
\frm{Velocità max moto armonico}{v_{\text{max}} = \omega A}
\frm{Accelerazione max moto armonico}{a_\text{max} = \omega^2 A}
\frm{Pendolo semplice}{\sum F_x = -m g \sin \theta = m a \implies a = -g \sin \theta}
\frm{Piccole oscillazioni pendolo}{2 \pi \sqrt{\frac{l}{g}}}
\frm{Trazione}{T = -\vec{P} + \vec{F}_\text{risalita} = mg + m \frac{v^2}{l}}

\subsection{Lavoro ed Energia}
\frmheader{Nome Formula}{Formula}
\frm{Lavoro}{L = \int_{\vec{r}_i}^{\vec{r}_f} \vec{F} \ d\vec{r} = \frac{1}{2} m v_f^2 - \frac{1}{2} m v_i^2 = F \frac{\Delta x}{\sin(\Theta)} \cos(\frac{\pi}{2} - \Theta)}
\frm{Energia cinetica}{K = \frac{1}{2} m v^2}
\frm{Energia potenziale}{\text{d}U = -F \text{d}x}
\frm{Energia meccanica}{E = U + K}
\frm{Energia potenziale peso}{U(y) = mgy}
\frm{Energia potenziale elastica}{U(x) = \frac{1}{2} k x^2}
\frm{Energia meccanica oscillatore armonico}{E = K + U = \frac{1}{2} m v^2 + \frac{1}{2} k x^2}
\frm{(in equilibrio)}{U(0) = 0 \text{, } K = E}
\frm{(al massimo)}{U(A) = E \text{, } K = 0 \text{, con A ampiezza}}
\frm{Potenza}{P = \frac{dL}{dt}}
\frm{Potenza su punto materiale}{P = \vec{F} \dot \vec{v}}


\section{Elettrostatica}
\frmheader{Nome Formula}{Formula}
\frm{Carica oggetto}{q = (N_p - N_e)e}
\frm{Costante di proporzionalità}{K = \frac{1}{4 \pi \epsilon_0 [\epsilon_r]} \text{ con $\epsilon_r$ solo non nel vuoto}}
\frm{Coulomb, cariche puntiformi}{\vec{F_{ab}} = \frac{1}{4 \pi \epsilon_0} \frac{q_a q_b}{r^2} \hat{r}_{ab} \text{ , } \vec{E} = \frac{1}{4 \pi \epsilon_0} \frac{q}{r^2} \hat{r}}
\frm{Campo da Forza}{\vec{E} = \frac{\vec{F}}{q}}
\frm{Accelerazione da campo}{a = \frac{q\vec{E}}{m}}
\frm{Gauss}{\Phi(\vec{E}) = \oint_A \vec{E} \cdot d\vec{A} = \frac{Q}{\epsilon_0}}
\frm{Potenziale per carica}{V = \frac{1}{4 \pi \epsilon_0} \frac{q}{r}}
\frm{Energia potenziale elettrica}{U = V q}
\frm{Capacità}{C = \frac{Q}{V}}
\frm{Condensatore piano}{C = \frac{[\epsilon_r] \epsilon_0 A}{d}}
\frm{Condensatori in serie}{\frac{1}{C} = \sum_i \frac{1}{  C_i}}
\frm{Condensatori in parallelo}{C = \sum_i C_i}
\frm{Energia potenziale condensatore}{U = \frac{Q^2}{2C} = \frac{1}{2} C V^2 = \frac{QV}{2}}
\frm{Densità di energia}{u = \frac{U}{A d} = \frac{1}{2} \epsilon_0 E^2}
\frm{Intensità di corrente \newline \scriptsize{$n$ densità portatori, $S$ sezione filo, $v_d$ velocità deriva}}{I = \frac{dQ}{dt} = n S v_d |q|}
\frm{Densità volumica}{n = \frac{dN}{dV} \text{, con $N$ numero di particelle e $V$ volume}}
\frm{Densità corrente}{J = \frac{I}{S} = n q v_d}
\frm{Legge di Ohm}{V = R I}
\frm{Resistività materiale}{R = \rho \frac{l}{S}}
\frm{Ohm con densità corrente}{\vec{E} = \rho \vec{J}}
\frm{Drude accelerazione con campo}{\vec{a} = - \frac{e \vec{E}}{m} = \frac{\vec{F}}{m}}
\frm{Drude velocità deriva}{\vec{v_d} = -\frac{e E}{m} \tau \text{ con $\tau$ tempo medio tra urti}}
\frm{Drude resistività}{\rho = \frac{m}{n e^2 \tau}}
\frm{Resistenze in serie}{R_{tot} = R_1 + R_2}
\frm{Resistenze in parallelo}{\frac{1}{R_{tot}} = \frac{1}{R_1} + \frac{1}{R_2}}
\frm{Legge di Joule}{P = V I = R I^2 = \frac{V^2}{R}}
\frm{Energia dissipata in resistenza}{\Delta U = -V \Delta Q}
\frm{1° Legge Kirchoff}{I_{in_1} + I_{in_2} = I_{out_1} + I_{out_2}}
\frm{2° Legge Kirchoff}{V - R_1 I - R_2 I - R_3 I = 0}
\frm{Carica condensatore}{q(t) = C \Delta V ( 1 - e ^ {- \frac{t}{RC}})}


\section{Magnetismo}
\frmheader{Nome Formula}{Formula}
\frm{Campo generato da un filo}{B \propto \frac{i}{r} \text{, con $i$ corrente nel filo, $r$ distanza dal filo}}
\frm{Legge di Lorentz}{\vec{F} = q \vec{V} \cdot \vec{B}}
\frm{Lorents con campo E}{\vec{F} = q \vec{E} + q (\vec{v} \cdot \vec{B})}
\frm{Modulo Lorenz}{F = |q| v B \sin \theta \text{, con $\theta$ angolo tra velocità e campo}}
\frm{Moto campo magnetico uniforme}{r = \frac{m v}{q B} \text{, } \omega_C = \frac{q B}{m} \text{ (detta frequenza di ciclotrone), } \nu = \frac{\omega_C}{2 \pi}}
\frm{Velocità in ciclotrone}{v = \frac{E}{B}}
\frm{Thomson}{\frac{e}{m} = \frac{y E}{B^2 (LD + \frac{L^2}{2})}}
\frm{Campo di Hall}{E_H = \frac{J B}{n q} = v_d B}
\frm{Coefficiente di Hall}{R_H = \frac{E_H}{JB} = \frac{1}{n q}}
\frm{Filo rettilineo}{F = I l \times B}
\frm{Momento meccanico torcente}{\tau = I S \times B}
\frm{Teorema di Ampére}{\oint B \ dr = \mu_0 \sum I_{conc}}
\frm{Filo dritto infinito}{B = \frac{\mu_0 I}{2 \pi R}}
\frm{Interno del filo}{B = \frac{\mu_0 I R}{2 \pi a^2}}
\frm{Solenoide}{B = \mu_0 n I \text{ all'interno, all'esterno è nullo}}
\frm{Toroide (ciambella)}{B = \frac{\mu_0 I N}{2 \pi R} \text{, con $N$ numero spire}}
\frm{Forza tra fili paralleli}{F = \frac{\mu_0 I_1 I_2 l}{2 \pi R}}
\frm{1° Laplace}{dB = \frac{\mu_0 I d l \times \hat{r}}{4 \pi r^2}}
\frm{1° Laplace con 1 spira}{B = \frac{\mu_0 I}{2 R}}
\frm{Ferromagnetismo}{B = \mu_r \mu_0 n I \text{, con $\mu_r$ la permeabilità del materiale}}


\section{Magnetismo 2}
\frmheader{Nome Formula}{Formula}
\frm{Flusso}{\Phi(B) = B S \cos \theta}
\frm{Faraday (Forza Elettromotrice Indotta)}{\mathcal{E}_{ind} = - \frac{d}{dt} (\Phi(B)) \text{, il flusso cambia nel tempo per $B$, $S$ o $\theta$}}
\frm{Corrente da F.E.M. con Ohm}{I = \frac{| \mathcal{E}_{ind} |}{R}}
\frm{Verso corrente}{\text{Se fuori \textbf{aumenta} viene \textbf{respinto}} \\ & \text{se fuori \textbf{diminuisce} viene \textbf{aiutato}}}
\frm{Induzione di moto}{\mathcal{E}_{ind} = B L_\text{filo} v_\text{filo}}
\frm{Velocità limite}{v_\text{max} = \frac{m g R}{B^2 L_\text{filo}^2}}
\frm{F.E.M. Alternatore}{\mathcal{E}_\text{ind} = N B S \omega \sin(\omega t) \text{, con $S$ area, N spire, ruota a velocità $\omega$}}
\frm{}{\mathcal{E}_\text{max} = N B S \omega \text{ (ovvero l'Ampiezza) }}
\frm{F.E.M. Trasformatore}{\frac{V_P}{V_S} = \frac{N_P}{N_S} \text{, } V_P I_P = V_S I_S}
\frm{Corrente di Spostamento (Ampère-Maxwell)}{I_s = \epsilon_0 \frac{d}{dt} \Phi (E)}
